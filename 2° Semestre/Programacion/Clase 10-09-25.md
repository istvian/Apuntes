## Relación de asociación

Programar ejemplo:

```python
# Cliente.py
from Factura import Factura

class Cliente:
    def __init__(self, rutCliente, correo):
        self.rut = rutCliente
        self.email = correo
        self.facturas: list[Factura] = []

    def agregarFact(self, unaFact):
        self.facturas.append(unaFact)

    def mostrarFact(self):
        print("Núm.Fact. \tFecha Fact.\tTotal Fact.")
        for fila in self.facturas:
            print(fila.numFactura, "\t\t", fila.fecha, "\t", fila.total)

```

```python
# Factura.py
from Cliente import *

class Factura:
    def __init__(self, numFactura, fecha, total, clienteDestino: Cliente):
        self.numFactura = numFactura
        self.fecha = fecha
        self.total = total
        self.clienteDestino = clienteDestino

```

```python
# Compras.py

from Factura import *
from Cliente import *

rutDeCliente = input("Ingrese rut del cliente empresa=")
email = input("Ingrese email=")
cli1 = Cliente(rutDeCliente, email)
cantFact = int(input("Cantidad de Facturas="))
for i in range(cantFact):
    numF = int(input("Número de fact." + str(i + 1) + "="))
    fechaF = input("Fecha de fact." + str(i + 1) + "=")
    totalF = float(input("Total de fact." + str(i + 1) + "="))

    fact1 = Factura(numF, fechaF, totalF, cli1)
    cli1.agregarFact(fact1)

cli1.mostrarFact()

```

## Relaciones de herencia

```python
# Casa.py
from Inmueble import *


class Casa(Inmueble):
    def __init__(self, direccion, superficieConst, superfPatio):
        super().__init__(direccion, superficieConst)
        self.superfPatio = superfPatio

```
