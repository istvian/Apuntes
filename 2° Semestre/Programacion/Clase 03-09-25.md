### Ejemplo

Se creara una clase para manejar cuentas bancarias, con los atributos de instancia numCuenta, tipoCuenta, saldo y rutCliente

3 operaciones:
depositar(), girar(), mostrarSaldo()

```python

class CuentaBancaria:

    def __init__(self, numCuenta, tipoCuenta, saldo, rutCliente):
        self.numCuenta = numCuenta
        self.tipoCuenta = tipoCuenta
        self.saldo = saldo
        self.rutCliente = rutCliente

    def depositar(self, monto):
        self.saldo += monto
        print("Se depositaron $", monto)
        return self

    def girar(self, monto):
        if self.saldo >= monto:
            self.saldo -= monto
            print("Se giraron $", monto)
        else:
            print("Monto ${} supera el saldo actual: ${}".format(monto, self.saldo))
        return self

    def mostrarSaldo(self):
        print("Cliente: {} | Saldo actual: ${}".format(self.rutCliente, self.saldo))
        return self

    def __str__(self):
        return (
            "Esta es la cuenta "
            + self.tipoCuenta
            + " número "
            + str(self.numCuenta)
            + " del cliente "
            + self.rutCliente
        )


cuenta = CuentaBancaria(4500989, "De Ahorro", 0, "20378956-9")

while True:
    print("\033c", end="")  # limpia terminal
    print("Menú")
    print(15 * "-")
    print("Depositar [d]")
    print("Girar [g]")
    print("Mostrar saldo [m]")
    print("Fin [f]")
    print(15 * "-")
    opc = input("Ingrese opción=").lower()
    if opc == "d":
        monto = int(input("¿Cuanto desea depositar? ="))
        cuenta.depositar(monto)
    elif opc == "g":
        monto = int(input("¿Cuanto desea girar? ="))
        cuenta.girar(monto)
    elif opc == "m":
        cuenta.mostrarSaldo()
    elif opc == "f":
        print("Cerrando programa")
        exit(0)
    else:
        print("Código equivocado")

    input("Presione una tecla para continuar")


```

class Herramienta:
    def __init__(self, codHerr, nombre, precio, stock):
        self.codHerr = codHerr
        self.nombre = nombre
        self.precio = precio
        self.stock = stock

    def vender(self, cantidad):
        if self.stock > cantidad:
            print("Total venta =$", self.precio * cantidad)
            self.stock -= cantidad
            print("Stock restante: ", self.stock)
        else:
            print("No hay stock suficiente, quedan ", self.stock)

    def aumentarStock(self, cantidad):
        if cantidad > 0:
            self.stock += cantidad
            print(
                "Stock aumentado en {}, stock actual: {}".format(cantidad, self.stock)
            )
        else:
            print("Ingresa una cantidad valida")

    def mostrarStock(self):
        print("Stock actual: {}".format(self.stock))

    print("\033c", end="")
