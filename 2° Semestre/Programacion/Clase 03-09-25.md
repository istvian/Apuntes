### Ejemplo

Se creara una clase para manejar cuentas bancarias, con los atributos de instancia numCuenta, tipoCuenta, saldo y rutCliente

3 operaciones:
depositar(), girar(), mostrarSaldo()

```python

class CuentaBancaria:

    def __init__(self, numCuenta, tipoCuenta, saldo, rutCliente):
        self.numCuenta = numCuenta
        self.tipoCuenta = tipoCuenta
        self.saldo = saldo
        self.rutCliente = rutCliente

    def depositar(self, monto):
        self.saldo += monto
        print("Se depositaron $", monto)
        return self

    def girar(self, monto):
        if self.saldo >= monto:
            self.saldo -= monto
            print("Se giraron $", monto)
        else:
            print("Monto ${} supera el saldo actual: ${}".format(monto, self.saldo))
        return self

    def mostrarSaldo(self):
        print("Cliente: {} | Saldo actual: ${}".format(self.rutCliente, self.saldo))
        return self

    def __str__(self):
        return (
            "Esta es la cuenta "
            + self.tipoCuenta
            + " n√∫mero "
            + str(self.numCuenta)
            + " del cliente "
            + self.rutCliente
        )


cuenta = CuentaBancaria(4500989, "De Ahorro", 0, "20378956-9")
print(cuenta)
cuenta.depositar(10000)
cuenta.mostrarSaldo()

cuenta.mostrarSaldo()

```
