# 1) 3 ESTRUCTURAS CICLICAS.....

codigos_cursos = []
cupos_cursos = []
precios_cursos = []
contador = 1
cantidad_cursos = 8

for i in range(contador, cantidad_cursos + 1):
    codigo = input("Ingresa código curso N°{}=".format(i))
    codigos_cursos.append(codigo)

# REINICIA CONTADOR EN 1
contador = 1
while True:
    try:
        cupos = int(input("Ingresa cupo curso N°{}=".format(contador)))
        while cupos <= 0:
            cupos = int(input("Error: Ingresa cupo curso N°{}=".format(contador)))
        cupos_cursos.append(cupos)
        if contador < cantidad_cursos:
            contador += 1
        else:
            break
    except ValueError:
        print("Error: Ingresa solo números mayores a 0")

# REINICIA CONTADOR EN 1
contador = 1
while True:
    try:
        precio = int(input("Ingresa precio curso N°{}=".format(contador)))
        while precio <= 0:
            precio = int(input("Error: Ingresa precio curso N°{}=".format(contador)))
        precios_cursos.append(precio)
        if contador < cantidad_cursos:
            contador += 1
        else:
            break
    except ValueError:
        print("Error: Ingresa solo números mayores a 0")


# 2) MOSTRAS LAS 3 LISTAS
print("=== Mostrar 3 listas ===")
print(codigos_cursos)
print(cupos_cursos)
print(precios_cursos)
print("\n")

print("El mayor de los {} cupos es: {}\n".format(cantidad_cursos, max(cupos_cursos)))

suma_precios = round(sum(precios_cursos), 2)  # MUESTRA CON 2 DECIMALES
print("La suma de los precios (con dos decimales) es: {}\n".format(suma_precios))


# 4) SI LA SUMA DE PRECIOS....
if suma_precios > 10000000:
    print("Superior")

    if suma_precios > 6000000 and suma_precios < 10000000:

lista_anidada = [codigos_cursos[:], cupos_cursos[:], precios_cursos[:]]

precio_nuevo = input("Ingrese un nuevo precio=")
