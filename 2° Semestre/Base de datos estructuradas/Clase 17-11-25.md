# Ejercicio grupal


ARCHIVO SYSTEM

```sql

-- A) CREAR TABLAS
create table clientesEmpresa( 
rutCliEmpresa varchar2(12) not null primary key, 
nomCliempresa varchar2(20), 
fonoCliEmpresa varchar2(14), 
emailCliEmpresa varchar2(20) 
); 

create table articulos( 
codArt varchar2(6) not null primary key, 
nomArt varchar2(20),  
precioArt number(8), 
fechFabrArt date, 
stockArt number(6) 
); 

create table ventasnacionales( 
numVenta number(8) not null primary key, 
fechaVenta date, 
montoVenta number(8), 
rutCliente varchar2(12) not null references clientesempresa, 
codArticulo varchar2(6)not null references articulos 
); 

--B) CREAR LOS SIGUIENTES USUARIOS Y ASIGNARLES LOS ROLES NECESARIOS
CREATE USER adminVentas identified by 123 default tablespace users;
CREATE USER adminBodega identified by 123 default tablespace users;

-- ASIGNAR PERMISO DE CONEXION
grant connect to adminVentas;
grant connect to adminBodega;

-- CREAR ROLES
CREATE ROLE ADMINVENTASROLE;
CREATE ROLE ADMINBODEGAROLE;

-- ASIGNAR PERMISOS SOBRE TABLAS
GRANT SELECT, INSERT, UPDATE, DELETE ON CLIENTESEMPRESA TO ADMINVENTASROLE;
GRANT SELECT, INSERT, UPDATE, DELETE ON VENTASNACIONALES TO ADMINVENTASROLE;

GRANT SELECT, INSERT, UPDATE, DELETE ON ARTICULOS TO ADMINBODEGAROLE;

-- ASIGNAR ROLE A USUARIO
GRANT ADMINVENTASROLE TO ADMINVENTAS;
GRANT ADMINBODEGAROLE TO ADMINBODEGA;
COMMIT;
```
